# Rust Integration

This guide shows how to integrate versionator with a Rust project.

## Project Structure

```
myproject/
├── VERSION
├── .versionator.yaml
├── Cargo.toml
├── Makefile
├── build.rs
└── src/
    ├── main.rs
    └── version.rs          # Optional: generated
```

## Setup

### Initialize Version

```bash
cd myproject
versionator version
# Creates VERSION with 0.0.0

# Enable prefix
versionator prefix set v
# VERSION: v0.0.0
```

## Approach 1: Build Script (Recommended)

Use a `build.rs` script to inject version at compile time.

### build.rs

```rust
// build.rs
use std::process::Command;

fn main() {
    // Get version from versionator
    let output = Command::new("versionator")
        .arg("version")
        .output()
        .expect("Failed to run versionator");

    let version = String::from_utf8(output.stdout)
        .expect("Invalid UTF-8")
        .trim()
        .to_string();

    println!("cargo:rustc-env=VERSION={}", version);

    // Get additional info
    let commit = Command::new("versionator")
        .args(["version", "-t", "{{ShortHash}}"])
        .output()
        .map(|o| String::from_utf8(o.stdout).unwrap_or_default())
        .unwrap_or_default()
        .trim()
        .to_string();

    println!("cargo:rustc-env=GIT_COMMIT={}", commit);

    let date = Command::new("versionator")
        .args(["version", "-t", "{{BuildDateTimeUTC}}"])
        .output()
        .map(|o| String::from_utf8(o.stdout).unwrap_or_default())
        .unwrap_or_default()
        .trim()
        .to_string();

    println!("cargo:rustc-env=BUILD_DATE={}", date);

    // Rerun if VERSION changes
    println!("cargo:rerun-if-changed=VERSION");
}
```

### main.rs

```rust
// src/main.rs
const VERSION: &str = env!("VERSION");
const GIT_COMMIT: &str = env!("GIT_COMMIT");
const BUILD_DATE: &str = env!("BUILD_DATE");

fn main() {
    println!("MyApp {} (commit: {}, built: {})",
        VERSION, GIT_COMMIT, BUILD_DATE);
}
```

## Approach 2: Generated File

Generate a Rust source file with version info.

```bash
versionator emit rust --output src/version.rs
```

Generated content:

```rust
// Auto-generated by versionator
pub const VERSION: &str = "1.2.3";
pub const VERSION_MAJOR: u32 = 1;
pub const VERSION_MINOR: u32 = 2;
pub const VERSION_PATCH: u32 = 3;
```

### Use in Code

```rust
// src/main.rs
mod version;

fn main() {
    println!("Version: {}", version::VERSION);
}
```

### Add to .gitignore

```
src/version.rs
```

## Cargo.toml Integration

### Version Field

You can automate Cargo.toml version updates:

```bash
# Update Cargo.toml version
VERSION=$(versionator version)
sed -i "s/^version = \".*\"/version = \"$VERSION\"/" Cargo.toml
```

Or use a script:

```bash
#!/bin/bash
# scripts/sync-version.sh
VERSION=$(versionator version | sed 's/^v//')  # Remove v prefix if present
sed -i "s/^version = \".*\"/version = \"$VERSION\"/" Cargo.toml
```

## Build Automation

### Makefile

```makefile
VERSION := $(shell versionator version)

.PHONY: build test release clean

build:
	cargo build --release

test:
	cargo test

# Using generated file approach
generate-version:
	versionator emit rust --output src/version.rs

# Sync Cargo.toml
sync-version:
	@VERSION=$$(versionator version | sed 's/^v//'); \
	sed -i "s/^version = \".*\"/version = \"$$VERSION\"/" Cargo.toml

release: sync-version
	versionator patch increment
	$(MAKE) sync-version
	git add VERSION Cargo.toml
	git commit -m "Release $(VERSION)"
	versionator tag
	git push --tags
	cargo publish

clean:
	cargo clean
```

### Just

```just
version := `versionator version`

# Build
build:
    cargo build --release

# Test
test:
    cargo test

# Generate version file (if using that approach)
emit-version:
    versionator emit rust --output src/version.rs

# Sync Cargo.toml version
sync-version:
    #!/bin/bash
    VERSION=$(versionator version | sed 's/^v//')
    sed -i "s/^version = \".*\"/version = \"$VERSION\"/" Cargo.toml

# Release
release bump="patch":
    versionator {{bump}} increment
    just sync-version
    git add VERSION Cargo.toml
    git commit -m "Release $(versionator version)"
    versionator tag
    git push --tags
    cargo publish
```

## CI/CD Integration

### GitHub Actions

```yaml
name: Build and Publish

on:
  push:
    tags:
      - 'v*'

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Install Rust
        uses: actions-rs/toolchain@v1
        with:
          toolchain: stable
          override: true

      - name: Install versionator
        run: go install github.com/benjaminabbitt/versionator@latest

      - name: Build
        run: cargo build --release

      - name: Test
        run: cargo test

      - name: Upload artifact
        uses: actions/upload-artifact@v3
        with:
          name: myapp
          path: target/release/myapp

  publish:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Install Rust
        uses: actions-rs/toolchain@v1
        with:
          toolchain: stable
          override: true

      - name: Publish to crates.io
        run: cargo publish
        env:
          CARGO_REGISTRY_TOKEN: ${{ secrets.CRATES_IO_TOKEN }}
```

## CLI Version Flag

Using [clap](https://github.com/clap-rs/clap):

```rust
use clap::Parser;

const VERSION: &str = env!("VERSION");

#[derive(Parser)]
#[command(version = VERSION)]
struct Cli {
    #[command(subcommand)]
    command: Commands,
}

#[derive(clap::Subcommand)]
enum Commands {
    // ...
}

fn main() {
    let cli = Cli::parse();
    // ...
}
```

## Cross-Compilation

Build for multiple targets:

```bash
# Add targets
rustup target add x86_64-unknown-linux-gnu
rustup target add x86_64-apple-darwin
rustup target add x86_64-pc-windows-gnu

# Build
VERSION=$(versionator version)
cargo build --release --target x86_64-unknown-linux-gnu
cargo build --release --target x86_64-apple-darwin
cargo build --release --target x86_64-pc-windows-gnu
```
