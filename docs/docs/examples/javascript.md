# JavaScript/TypeScript Integration

This guide shows how to integrate versionator with JavaScript and TypeScript projects.

## Project Structure

```
myproject/
├── VERSION
├── .versionator.yaml
├── package.json
├── Makefile
└── src/
    ├── index.ts
    └── version.ts          # Generated
```

## Setup

### Initialize Version

```bash
cd myproject
versionator version
# Creates VERSION with 0.0.0
```

## Generate Version File

### JavaScript

```bash
versionator emit js --output src/version.js
```

Generated content:

```javascript
// Auto-generated by versionator
export const VERSION = "1.2.3";
export const VERSION_INFO = {
  major: 1,
  minor: 2,
  patch: 3,
};
```

### TypeScript

```bash
versionator emit ts --output src/version.ts
```

Generated content:

```typescript
// Auto-generated by versionator
export const VERSION = "1.2.3";
export const VERSION_INFO = {
  major: 1,
  minor: 2,
  patch: 3,
} as const;
```

## Package Integration

### Import in Code

```typescript
// src/index.ts
import { VERSION, VERSION_INFO } from './version';

console.log(`App version: ${VERSION}`);
console.log(`Major: ${VERSION_INFO.major}`);
```

### Sync with package.json

Keep `package.json` version in sync:

```bash
# Read version from VERSION file
VERSION=$(versionator version)

# Update package.json
npm version $VERSION --no-git-tag-version
```

Or use [npm-version-from-file](https://www.npmjs.com/package/version-from-file):

```json
{
  "scripts": {
    "version-sync": "npm version $(cat VERSION) --no-git-tag-version"
  }
}
```

## Build Automation

### package.json Scripts

```json
{
  "scripts": {
    "version:emit": "versionator emit ts --output src/version.ts",
    "version:sync": "npm version $(versionator version) --no-git-tag-version",
    "prebuild": "npm run version:emit",
    "build": "tsc",
    "bump:patch": "versionator patch increment && npm run version:sync",
    "bump:minor": "versionator minor increment && npm run version:sync",
    "bump:major": "versionator major increment && npm run version:sync"
  }
}
```

### Makefile

```makefile
VERSION := $(shell versionator version)

.PHONY: version build test publish

version:
	versionator emit ts --output src/version.ts
	npm version $(VERSION) --no-git-tag-version

build: version
	npm run build

test: version
	npm test

publish: build
	npm publish

release:
	versionator patch increment
	$(MAKE) version
	git add VERSION package.json src/version.ts
	git commit -m "Release $(VERSION)"
	versionator tag
	git push --tags
	$(MAKE) publish
```

### Just

```just
version := `versionator version`

# Emit version file
emit-version:
    versionator emit ts --output src/version.ts
    npm version {{version}} --no-git-tag-version

# Build
build: emit-version
    npm run build

# Test
test: emit-version
    npm test

# Release
release bump="patch":
    versionator {{bump}} increment
    just emit-version
    git add VERSION package.json src/version.ts
    git commit -m "Release $(versionator version)"
    versionator tag
    git push --tags
    npm publish
```

## CI/CD Integration

### GitHub Actions

```yaml
name: Publish Package

on:
  push:
    tags:
      - 'v*'

jobs:
  publish:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          registry-url: 'https://registry.npmjs.org'

      - name: Install versionator
        run: go install github.com/benjaminabbitt/versionator@latest

      - name: Generate version file
        run: |
          versionator emit ts --output src/version.ts
          npm version $(versionator version) --no-git-tag-version

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: npm run build

      - name: Publish
        run: npm publish
        env:
          NODE_AUTH_TOKEN: ${{ secrets.NPM_TOKEN }}
```

## Framework-Specific

### React

```typescript
// src/version.ts is generated

// In your app
import { VERSION } from './version';

function App() {
  return (
    <footer>
      <p>Version: {VERSION}</p>
    </footer>
  );
}
```

### Next.js

Use environment variables in `next.config.js`:

```javascript
// next.config.js
const { execSync } = require('child_process');

const VERSION = execSync('versionator version').toString().trim();

module.exports = {
  env: {
    APP_VERSION: VERSION,
  },
};
```

```typescript
// In your app
const version = process.env.APP_VERSION;
```

### Node.js CLI

```typescript
#!/usr/bin/env node
import { VERSION } from './version';
import { Command } from 'commander';

const program = new Command();

program
  .version(VERSION)
  .description('My CLI application');

program.parse();
```

## Custom Template

```bash
# Dump default template
versionator emit dump ts > version.tmpl
```

Edit `version.tmpl`:

```typescript
/**
 * {{AppName}} version information
 * Generated by versionator - do not edit
 */

export const VERSION = "{{MajorMinorPatch}}";
export const BUILD_DATE = "{{BuildDateTimeUTC}}";
export const GIT_COMMIT = "{{ShortHash}}";

export const VERSION_INFO = {
  major: {{Major}},
  minor: {{Minor}},
  patch: {{Patch}},
  buildDate: "{{BuildDateTimeUTC}}",
  gitCommit: "{{ShortHash}}",
} as const;
```

Use custom template:

```bash
versionator emit --template-file version.tmpl \
    --output src/version.ts \
    --set AppName="MyApp"
```
